Workflow con Grunt

Organizaremos las carpetas en dos grandes directorios:

  • src: donde guardaremos las fuentes y que será nuestro directorio de trabajo.
  • dist: donde Grunt compilará las fuentes y del cual nosotros no tocaremos nada.

clean

grunt-contrib-clean


clean: {
    build: [
        'dist'
    ]
}

autoprefixer

grunt-autoprefixer


autoprefixer: {
    options: {
        browsers: [
            'last 2 versions',
            '> 2%',
            'ie 8',
            'ie 9'
        ]
    },
    dist: {
        src: 'dist/css/main.css',
        dest: 'dist/css/main.css'
    }
},

pixrem

grunt-pixrem


pixrem: {
    options: {
        rootvalue: '100%'
    },
    dist: {
        src: 'dist/css/main.css',
        dest: 'dist/css/main.css'
    }
},

concat

grunt-contrib-concat


concat: {
    dist: {
        src: [
            'src/js/components/**/*.js'
        ],
        dest: 'dist/js/main.js'
    }
},

uglify

grunt-contrib-uglify


uglify: {
    main: {
        files: {
            'dist/js/main.min.js': 'dist/js/main.js'
        }
    }
},

copy

grunt-contrib-copy


 copy: {
    main: {
        expand: true,
        cwd: 'src/',
        src: '**/*.html',
        dest: 'dist/',
        filter: 'isFile',
    },
},
